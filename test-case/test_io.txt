// Test 10: Read/Write Operations
func ioTest() => void {
    local input : integer;
    local output : float;
    
    read(input);
    output := input * 2.5;
    write(output);
    
    if (input > 0) then {
        write(input + 100);
    } else {
        read(input);
        write(-input);
    };
}
