// Test 11: Complex Arithmetic Expressions
func arithmeticTest() => float {
    local a : integer;
    local b : float;
    local result : float;
    
    a := 10;
    b := 3.14;
    
    result := +a + -b * (5 + 3) / 2 - 1;
    result := result + a * b - (a + b) * 2;
    result := not (a > b) and (a <= 10 or b >= 3.0);
    
    return (result);
}
