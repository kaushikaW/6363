// Test 7: Array Operations
class ArrayTest {
    private attribute matrix : integer[10][20];
    private attribute vector : float[];
    
    public func processArrays() => void {
        local temp : integer[5];
        local index : integer;
        
        index := 0;
        temp[index] := matrix[1][2] + 5;
        vector[index + 1] := 3.14;
        
        while (index < 5) {
            temp[index] := temp[index] * 2;
            index := index + 1;
        };
    }
};
