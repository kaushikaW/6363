func operatorTest() => void {
    local a : integer;
    local b : integer;
    local c : float;
    
    // Assignment
    a := 10;
    
    // Arithmetic operators
    a := a + b - 5;
    a := a * b / 2;
    c := +a + -b;
    
    // Relational operators
    if (a == b) then { write(1); } else { };
    if (a <> b) then { write(2); } else { };
    if (a < b) then { write(3); } else { };
    if (a > b) then { write(4); } else { };
    if (a <= b) then { write(5); } else { };
    if (a >= b) then { write(6); } else { };
    
    // Logical operators
    if (a > 0 and b > 0) then { write(7); } else { };
    if (a > 0 or b > 0) then { write(8); } else { };
    if (not (a == 0)) then { write(9); } else { };
}
